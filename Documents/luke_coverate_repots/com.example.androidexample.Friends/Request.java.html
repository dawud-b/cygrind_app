package com.example.androidexample.Friends;

import android.util.Log;

import com.example.androidexample.User;

import org.json.JSONException;
import org.json.JSONObject;

public class Request {
    private Long id;
    private User sender;
    private User recipient;

    public Request(User sender, User recipient) {
        this.sender = sender;
        this.recipient = recipient;
    }

    public Request(JSONObject request) throws JSONException {
        JSONObject senderObj, recipientObj;
        try {
            senderObj = request.getJSONObject("sender");
            recipientObj = request.getJSONObject("receiver");
            id = request.getLong("friendRequestId");
            this.sender = new User(senderObj);
            this.recipient = new User(recipientObj);
        } catch (JSONException e) {
            Log.e("Request JSON conversion failed", e.toString());
        }
    }
    public Long getId() {
        return id;
    }

    public String getSender() {
        return sender.getUsername();
    }

    public String getRecipient() {
        return recipient.getUsername();
    }
}
