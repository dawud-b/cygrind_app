 @Entity
 public class WorkoutDate {
 
     @Id

     private WorkoutTemplate workout;

     private User user;
 
     private boolean completed;
 
     @JsonFormat(pattern = "yyyy-MM-dd'T'HH:mm:ss.SSSXXX")
     private Calendar date;

         this.workout = workoutTemplate;
         this.date = date;
         this.completed = false;
     }

 }
