package com.example.androidexample.Messaging;

import android.util.Log;

import com.example.androidexample.User;

import org.json.JSONException;
import org.json.JSONObject;

/**
 * The Reaction class represents a reaction that a user can apply to a message in a messaging application.
 * A reaction is typically an emoji or other form of feedback on a message.
 * This class encapsulates the user's information, the type of reaction, and the message to which the reaction applies.
 * <p>
 * Reactions are represented as integers, with each integer corresponding to a specific type of reaction (e.g., like, dislike, love).
 */
public class Reaction {

    // Constants for different reaction types
    static final int LIKED = 0;   // Reaction type: "Liked"
    static final int DISLIKED = 1; // Reaction type: "Disliked"
    static final int LOVED = 2;    // Reaction type: "Loved"
    static final int LAUGHING = 3; // Reaction type: "Laughing"
    static final int STRONG = 4;   // Reaction type: "Strong"
    static final int COOL = 5;     // Reaction type: "Cool"
    static final int CRY = 6;      // Reaction type: "Cry"

    private Long id;       // Unique ID for the reaction
    private User user;     // User who applied the reaction
    private Message message; // Message to which the reaction is applied
    private int reactionType; // Type of reaction (e.g., LIKED, DISLIKED, etc.)

    /**
     * Constructs a new Reaction instance with the specified user and reaction type.
     *
     * @param user The user who is reacting.
     * @param reactionType The type of reaction (e.g., LIKED, DISLIKED, etc.).
     */
    public Reaction(User user, int reactionType) {
        this.user = user;
        this.reactionType = reactionType;
    }

    /**
     * Constructs a new Reaction instance by parsing a JSON object.
     * This constructor extracts the reaction type from the provided JSON object.
     *
     * @param reactionObj The JSON object representing the reaction.
     */
    public Reaction(JSONObject reactionObj) {
        try {
            reactionType = reactionObj.getInt("reactionType");
        } catch (JSONException e) {
            Log.e("Reaction", "JSONException parsing message reaction");
        }
    }

    /**
     * Gets the unique ID of the reaction.
     *
     * @return The ID of the reaction.
     */
    public Long getId() {
        return id;
    }

    /**
     * Gets the user who applied the reaction.
     *
     * @return The user who reacted.
     */
    public User getUser() {
        return this.user;
    }

    /**
     * Sets the user who applied the reaction.
     *
     * @param user The user to set as the one who reacted.
     */
    public void setUser(User user) {
        this.user = user;
    }

    /**
     * Gets the type of reaction (e.g., LIKED, DISLIKED, etc.).
     *
     * @return The reaction type.
     */
    public int getReactionType() {
        return reactionType;
    }

    /**
     * Sets the type of reaction (e.g., LIKED, DISLIKED, etc.).
     *
     * @param reactionType The reaction type to set.
     */
    public void setReactionType(int reactionType) {
        this.reactionType = reactionType;
    }

    /**
     * Gets the message to which the reaction applies.
     *
     * @return The message associated with this reaction.
     */
    public Message getMessage() {
        return this.message;
    }

    /**
     * Sets the message to which the reaction applies.
     *
     * @param message The message to set as the one associated with this reaction.
     */
    public void setMessage(Message message) {
        this.message = message;
    }
}
